182 template <typename T>
183 auto circular_buffer<T>::circ_inc(size_type base, difference_type inc, const circular_buffer<T>& buf) -> size_type
184 {
185  difference_type signed_new_base = base + inc;
186  const difference_type max_size = buf.entry_.size();
187 
188  // Adjust from the negative direction
189  while (signed_new_base < 0)
190  signed_new_base += max_size;
191 
192  // Adjust from the positive direction
193  while (signed_new_base >= max_size)
194  signed_new_base -= max_size;
195 
196  return static_cast<size_type>(signed_new_base);
197 }

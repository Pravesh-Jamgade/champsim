107 {

110 
111  unsigned int new_signature = 0;

115  new_signature = ((old_signature << SIG_SHIFT) ^ sig_delta) & SIG_MASK;

117  // printf("old_signature: %x SIG_SHIFT: %d sig_delta: %d SIG_LENGTH:
